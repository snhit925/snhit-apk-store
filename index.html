// Game Core Constants & Configurations const BOARD_SIZE = 15; const COLORS = ['red', 'green', 'yellow', 'blue']; // Global Engine State Tracker let gameState = { currentPlayerIndex: 0, currentDiceRoll: null, hasRolled: false, // Track coordinate position mappings: -1 means locked inside Base Home tokens: { red: [{ pos: -1, elementId: 'red-0' }], green: [{ pos: -1, elementId: 'green-0' }], yellow: [{ pos: -1, elementId: 'yellow-0' }], blue: [{ pos: -1, elementId: 'blue-0' }] } }; // Precise Track Sequence Map Array for Ludo Circuit Path Movement const commonTrackCoordinates = [ {r:6, c:1}, {r:6, c:2}, {r:6, c:3}, {r:6, c:4}, {r:6, c:5}, {r:5, c:6}, {r:4, c:6}, {r:3, c:6}, {r:2, c:6}, {r:1, c:6}, {r:0, c:6}, {r:0, c:7}, {r:0, c:8}, {r:1, c:8}, {r:2, c:8}, {r:3, c:8}, {r:4, c:8}, {r:5, c:8}, {r:6, c:9}, {r:6, c:10}, {r:6, c:11}, {r:6, c:12}, {r:6, c:13}, {r:6, c:14}, {r:7, c:14}, {r:8, c:14}, {r:8, c:13}, {r:8, c:12}, {r:8, c:11}, {r:8, c:10}, {r:8, c:9}, {r:9, c:8}, {r:10, c:8}, {r:11, c:8}, {r:12, c:8}, {r:13, c:8}, {r:14, c:8}, {r:14, c:7}, {r:14, c:6}, {r:13, c:6}, {r:12, c:6}, {r:11, c:6}, {r:10, c:6}, {r:9, c:6}, {r:8, c:5}, {r:8, c:4}, {r:8, c:3}, {r:8, c:2}, {r:8, c:1}, {r:8, c:0}, {r:7, c:0} ]; // Entry Indexes depending on standard Ludo rules rulesets const START_INDEX = { red: 0, green: 13, yellow: 26, blue: 39 }; // Initialize structural components dynamically function buildTrackCells() { const board = document.getElementById('board'); commonTrackCoordinates.forEach((coord, index) => { const cell = document.createElement('div'); cell.className = 'cell common-tile'; cell.style.gridRowStart = coord.r + 1; cell.style.gridColumnStart = coord.c + 1; cell.setAttribute('data-index', index); // Highlight Starting safe spots if(index === START_INDEX.red) cell.style.backgroundColor = 'var(--red)'; if(index === START_INDEX.green) cell.style.backgroundColor = 'var(--green)'; if(index === START_INDEX.yellow) cell.style.backgroundColor = 'var(--yellow)'; if(index === START_INDEX.blue) cell.style.backgroundColor = 'var(--blue)'; board.appendChild(cell); }); } // Dice Roll Calculations document.getElementById('roll-btn').addEventListener('click', () => { if (gameState.hasRolled) return; // Generate Random 1-6 Index gameState.currentDiceRoll = Math.floor(Math.random() * 6) + 1; const diceEl = document.getElementById('dice-element'); diceEl.innerText = gameState.currentDiceRoll; gameState.hasRolled = true; evaluateMovableTokens(); }); // Evaluate movement rules function evaluateMovableTokens() { const activeColor = COLORS[gameState.currentPlayerIndex]; const tokens = gameState.tokens[activeColor]; let possibleMoves = 0; tokens.forEach(token => { const el = document.getElementById(token.elementId); // Can release token from yard base if a 6 is rolled if (token.pos === -1 && gameState.currentDiceRoll === 6) { el.classList.add('movable'); possibleMoves++; } else if (token.pos >= 0) { el.classList.add('movable'); possibleMoves++; } }); // Automatically skip turn if zero functional legal maneuvers are present if (possibleMoves === 0) { setTimeout(passTurn, 1000); } } // Process Piece movement adjustments document.querySelectorAll('.token').forEach(tokenElement => { tokenElement.addEventListener('click', (e) => { if (!e.target.classList.contains('movable')) return; const color = e.target.getAttribute('data-color'); const id = parseInt(e.target.getAttribute('data-id')); const tokenData = gameState.tokens[color][id]; if (tokenData.pos === -1 && gameState.currentDiceRoll === 6) { tokenData.pos = START_INDEX[color]; } else { tokenData.pos = (tokenData.pos + gameState.currentDiceRoll) % commonTrackCoordinates.length; } renderTokenPositions(); clearMovableHighlights(); passTurn(); }); }); function clearMovableHighlights() { document.querySelectorAll('.token').forEach(el => el.classList.remove('movable')); } function passTurn() { gameState.currentPlayerIndex = (gameState.currentPlayerIndex + 1) % 4; gameState.hasRolled = false; const nextColor = COLORS[gameState.currentPlayerIndex].toUpperCase(); const indicator = document.getElementById('turn-indicator'); indicator.innerHTML = `Current Turn: ${nextColor}`; } function renderTokenPositions() { COLORS.forEach(color => { gameState.tokens[color].forEach(token => { const el = document.getElementById(token.elementId); if (token.pos === -1) { // Return to home container DOM element document.querySelector(`.${color}-base .base-inner`).appendChild(el); } else { // Append directly to active track cell const targetCell = document.querySelector(`.common-tile[data-index="${token.pos}"]`); if (targetCell) targetCell.appendChild(el); } }); }); } // Live runtime appearance modification linkages document.getElementById('bg-picker').addEventListener('input', (e) => { document.documentElement.style.setProperty('--board-bg', e.target.value); }); document.getElementById('border-picker').addEventListener('input', (e) => { document.documentElement.style.setProperty('--border-color', e.target.value); }); // Primary Entry Bootloader Execution buildTrackCells(); renderTokenPositions();
